function alphabetize(inputArray){
    var outputArray=[];
    while(inputArray.length){
        var indexOfSmallest=0;
        for(var i=1;i<inputArray.length;i++){
            if(isSmallerThanWordAtIndex(inputArray[i].toLowerCase(),inputArray[indexOfSmallest].toLowerCase())){
                indexOfSmallest=i;
            }
        }
        outputArray.push( inputArray.splice(indexOfSmallest,1)[0] );
    }

    return outputArray;

}
function isSmallerThanWordAtIndex(test,against){
    var word1=test.replace(/^.*class="drag" >/,"");
    word1=word1.replace(/<.*$/,"");
    var word2=against.replace(/^.*class="drag" >/,"");
    word2=word2.replace(/<.*$/,"");
    var isIt=[];
    isIt[isIt.length]=word1;
    isIt[isIt.length]=word2;
    //alert(isIt);
    isIt=isIt.sort();//alert(isIt);
    return isIt[0] == word1;
}
