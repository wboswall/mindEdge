<h4>The alt (alternative) attribute: (informative vs. decorative) images</h4>

<p>The <a role="button" tabindex="0" class="glossary-alert" id="glossTerm0" data-term="alt attribute" name="alt attribute" title="The alternative text in case an image can't be displayed">alt attribute<span class="screenreader">*</span></a><span style="display:none;">&nbsp;The alternative text in case an image can't be displayed&nbsp;</span> is text that describes the image. The purpose of alternative or alt text is twofold:</p> 
  
  <ul>
	<li>The text within the alt attribute is displayed if a browser can't locate the image.</li>
 
	<li>Alt text is also used by screen readers, so it's critical for accessibility. Without alt text, a user who relies on screen reader technology would be missing out on what the image adds to a sighted reader's understanding of the webpage content. Therefore, it is essential that alt text fully describes the image</li>
</ul>

<div class="big yellow highlightbox">
  <p><b>Note:</b> you must include the alt attribute with every image in your page, but it is permissible to leave the value empty (alt= " ") when the image is merely decorative and adds nothing to the understanding of the page's written content. The empty attribute alerts screen readers to ignore the image, so it doesn't add audible clutter to the page. If your pages are full of decorative images, however, you almost certainly have a fundamental design problem that is keeping you from helping your site visitors find and understand the content they need on your site without wading through the visual clutter of purely decorative images.</p>
</div>

class CodeBlockClass {

    constructor(blockIndex, type, debug) {
        this.container = '';
        this.index = '';
        this.options = {
            mode: "htmlmixed",
            styleActiveLine: false,
            lineNumbers: true,
            lineWrapping: true,
            theme: "rubyblue",
            autoRefresh: true,
            matchBrackets: true,
            showHint:false,
            indentWithTabs: true,
            autoCloseTags: {
                whenOpening: false,
                whenClosing: true
            },
            value: ''
        };
        let sectionTag = 'codeSection';
        this.index = blockIndex;
        switch (type) {
            case 'html':
                this.options.mode = 'htmlmixed';
                this.options.theme = 'rubyblue';
                sectionTag = 'htmlSection';
                break;
            case 'css':
                this.options.mode = type;
                this.options.theme = 'rubyblue';
                sectionTag = 'cssSection';
                break;
            case 'js':
                this.options.theme = 'rubyblue';
                this.options.mode = 'javascript';
                sectionTag = 'codeSection';
                break;
        }
        if(debug) {
            this.options.mode= 'html';
        }
        this.container = document.getElementById(`${sectionTag}-${blockIndex}`);
    }


    init() {
        this.editor = CodeMirror.fromTextArea(this.container, this.options);
        this.editor.setSize(null, 400);
    }

    getContent() {
        return this.editor.getValue();
    }

    setContent($content) {
        this.editor.setValue($content);
    }

    updateTheme(theme) {
        this.editor.setOption("theme", theme);
    }

   refresh() {
       this.editor.refresh();
    }

}

function resetCodeBlock(codeBlock) {
    if(codeBlock){
        Object.keys(codeBlock.blocks).forEach(type => {
            let content = $(`.${type}-reset-container-${codeBlock.section}`).val();
            if(type==='html'){
                const regex = /(<link\b.+href=")(.*[/[styles].css]*)(".*>)/gm;
               content = content.replace(regex, '<link rel="stylesheet" href="./styles.css"/>');
            }
            codeBlock.blocks[type].setContent(content);
        })
        codePreview(codeBlock)
    }
}

function toggleType(section, type){
    const sectionContainer =$(`#codingSection-${section}`);
    const htmlSection = sectionContainer.find(`#html`);
    const cssSection = sectionContainer.find(`#css`);
    const jsSection = sectionContainer.find(`#js`);
    if(lastType[section] === type){
        return;
    }
    lastType[section] = type;
    switch (type) {
        case 'html':
            htmlSection.toggleClass('active');
            cssSection.removeClass('active');
            jsSection.removeClass('active');
            break;
        case 'css':
            cssSection.toggleClass('active');
            htmlSection.removeClass('active');
            jsSection.removeClass('active');
            break;
        case 'js':
            jsSection.toggleClass('active');
            cssSection.removeClass('active');
            htmlSection.removeClass('active');
            break;
    }
    codeBlocks[section].blocks[type].refresh();
}


function updateFrame(section, html) {
    const previewFrame = document.getElementById(`preview-content-${section.section}`);
    const preview =  previewFrame.contentDocument ||  previewFrame.contentWindow.document;
    preview.open();
    preview.write(html);
    preview.close();
}

function replaceStyles(html, cssBlock){
    const regex = /(<link\b.+href=")(?!http)([/styles.css]*)(".*>)/gm;
    const cssContent = cssBlock.getContent();
    const content = `<style> ${cssContent} </style>`;
    html= html.replace(regex, content);
    return html;
}

function replaceJsScripts(html, jsBlock) {
    const regex = /(<script\b.+src=")(?!http)([/scripts.js]*)(".*>)/gm;
    const start ='<';
    const end=">";
    const jsContent = jsBlock.getContent();
    const content = `${start}script${end} ${jsContent} ${start}/script${end}`;
    html= html.replace(regex, content);
    return html;
}

function codePreview(section){
    let html = section.blocks.html.getContent();
    if(section.blocks.css){
        html = replaceStyles(html, section.blocks.css);
    }
    if(section.blocks.js){
        html = replaceJsScripts(html, section.blocks.js);
    }
    updateFrame(section, html);
}

function SaveContent(section) {
    const button = $(`#save-code-block-${section.section}`);
    const contentId = button.attr('data-content-id');
    const data = {
        contentId: contentId,
        sectionId: section.section,
        html: JSON.stringify(section.blocks['html'].getContent())
    };
    if(section.blocks['css']){
        data.css =JSON.stringify(section.blocks['css'].getContent());
    }
    if(section.blocks['js']){
        data.js =JSON.stringify(section.blocks['js'].getContent());
    }
    button.attr("disabled", true);
    return $.ajax({
        type: "POST",
        url: "/mods/code_block/codeBlockSave.php",
        dataType: "json",
        data: data,
        success: function(response) {
          console.log(response);
          button.removeAttr("disabled");
        }
    });
}

function updateTheme (section, theme) {
    if(section) {
        Object.keys(section.blocks).forEach($key => {
            section.blocks[$key].updateTheme(theme);
        });
    }
}
