/* helper function snippet to reshuffle aray, using Fisher-Yates algorithm */
shuffle = function(array) {
    var j, temp;
    for(var g = array.length-1; g >= 0; g--) {
        j = Math.floor(Math.random()*array.length);
        temp = array[j];
        array[j] = array[g];
        array[g] = temp;
    }
    return array;
};
