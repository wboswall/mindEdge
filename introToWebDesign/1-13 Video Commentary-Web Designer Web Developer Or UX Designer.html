    $("body").click(function(event){

        if(!inactivity_alerted){
            activity_pause_secs=0;
        }
        var $clicked = $(event.target);
        if($clicked.closest(".jwplayer").length){
            // Delegate pause to the player, which already pauses all other players on play.
            // As a result, we will not pause if someone clicks non-play controls on a different video,
            // but that is not really something we need to worry about
            event.stopPropagation();
        }else{
            // don't pause on opening of the transcript for the currently playing video
            if(!($clicked.is("a[href]") && $clicked.attr("href").indexOf("closedCaptionTranscript.php")>-1 && $clicked.attr("href").indexOf(window.Atutor.components.jwplayer.current)>-1)){
                if(window.Atutor.components.jwplayer.current) {
                    jwplayer(window.Atutor.components.jwplayer.current).pause(true);
                    window.Atutor.components.jwplayer.current = '';
                }
            }
        }
    });
